def start_index(sequence):
    n = len(sequence)
    for i in range(n):
        if sequence[i:i+3] == "AUG":

    return i

Step 3#

def translate_sequence(sequence):
    j = start_index(sequence)
    n = len(sequence)
    result = ""
    for i in range(j, n-3, 3):
        codon = sequence[i:i+3]
        if codon == "UAG" or codon == "UAA" or codon == "UGA":
            break
        else:
            result = result + translate(codon)
    return result
    

rna_seq = "GCAUAUGUUCAUAUGAAUA"
aa = translate_sequence(rna_seq)
print(aa)

Exercise 2: Letter Frequencies#

import numpy as np
import matplotlib.pyplot as plt

with open("english.txt") as f:
     text = f.read()

alphabet = "abcdefghijklmnopqrstuvwxyz"

counts = np.zeros(26) # this syntax creates a list containing 26 zeros

text = text.lower()

for c in text:
    i = alphabet.find(c)
    if i > -1:
        counts[i] += 1
        
plt.bar(list(alphabet), counts)
